Run Express as
Serverless Functions

No servers to manage. Just export your Express app and it runs on IGA Pages.

api/express/[[default]].js 
import express from 'express';

const app = express();
app.use(express.json());

app.get('/', (req, res) => {
  res.json({ message: 'Hello from Express on Node Functions!' });
});

// No app.listen() — just export
export default app;
